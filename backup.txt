same_exits = veiculos.groupby("Inicial").indices
for i in atendimentos:
    
    prioridade = vitimas[vitimas["Ponto"] == atendimentos[i]]["Prioridade"].values[0]
    maior_prioridade = prioridade
    veiculo_principal=i
    inicial = ordered_routes[i][0]
    
    for j in same_exits[inicial]:
        if j in atendimentos.keys():
            prioridade_j = vitimas[vitimas["Ponto"] == atendimentos[j]]["Prioridade"].values[0]
            if prioridade_j > maior_prioridade:
                maior_prioridade=prioridade_j
                veiculo_principal=j
    
    if veiculo_principal != i:
        for j in range(len(ordered_routes[veiculo_principal])-1):
            arco1 = (ordered_routes[veiculo_principal][j],ordered_routes[veiculo_principal][j+1])
            arco2 = (ordered_routes[i][j],ordered_routes[i][j+1])
            if arco1 == arco2:
                
                ordered_routes[i].remove(ordered_routes[i][j])
            else:
                break
